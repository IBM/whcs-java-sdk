          contentType = analyzeWithFlowOptions.contentType;
          returnAnalyzedText = analyzeWithFlowOptions.returnAnalyzedText;
      }

      /**
       * Instantiates a new builder.
       */
      public Builder() {
      }

      /**
       * Instantiates a new builder with required properties.
       *
